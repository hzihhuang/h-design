# getRandomColor

获取随机颜色

```tsx ｜ pure
import { getRandomColor } from 'HDesign';

getRandomColor(); // '#f0f0f0'
getRandomColor('hex'); // '#f0f0f0'
getRandomColor('rgb'); // 'rgb(255, 255, 255)'
getRandomColor('hsl'); // 'hsl(0, 0%, 100%)'
getRandomColor('hsv'); // 'hsv(0, 0%, 100%)'
```
